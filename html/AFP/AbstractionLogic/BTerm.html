Theory BTerm

theory BTerm imports Locales
begin

section ‹De Bruijn Term›

subsection ‹Terms›

datatype bterm = 
  FreeVar variable ‹bterm list›
| BoundVar nat
| Abstr abstraction ‹bterm list›

subsection ‹Free Atoms›

datatype atom =
  Var variable nat
| Unbound nat 

type_synonym atoms = "atom set"

context sigloc begin

fun freeAtomsAt :: "nat ⇒ bterm ⇒ atoms" where 
  "freeAtomsAt level (FreeVar x ts) = 
     (§fold atoms = {Var x (length ts)}, t=ts!_. atoms ∪ freeAtomsAt level t)"
| "freeAtomsAt level (BoundVar i) = 
     (if i < level then {} else {Unbound (i - level)})"
| "freeAtomsAt level (Abstr a ts) = 
     (§fold atoms = {}, t=ts!_. atoms ∪ freeAtomsAt (level + §v a) t)" 

definition freeAtoms :: "bterm ⇒ atoms" where
  "freeAtoms t = freeAtomsAt 0 t"

definition unboundAtoms :: "nat set ⇒ atoms" ("↑") where
  "↑ubs = { Unbound u | u. u ∈ ubs }"

subsection ‹Wellformedness›

fun bt_wf :: "bterm ⇒ bool" where
  "bt_wf (FreeVar x ts) = (∀ t=ts!_. bt_wf t)"
| "bt_wf (BoundVar i) = True"
| "bt_wf (Abstr a ts) = (✓a ∧ §a a = length ts ∧ 
     (∀ t=ts!i. bt_wf t ∧ freeAtoms t ∩ ↑(nats (§v a)) ⊆ ↑(a!♮i)))"

end

end
