Theory NTerm

theory NTerm imports Algebra 
begin

section ‹Term›

subsection ‹Variables›

type_synonym variable = string 

type_synonym variables = "(variable × nat) set"

definition binders_as_vars :: "variable list ⇒ variables" ("_`,0" [1000] 1000) where 
  "xs`,0 = { (x, 0) | x. x ∈ set xs }"

lemma binders_as_vars_empty[simp]: "[]`,0 = {}"
  by (simp add: binders_as_vars_def) 

lemma deduction_forall_deps_0[iff]: "𝔇!!abstr_forall.@0([x]) = [x]"
  apply (auto)
  apply (subst has_shape_get)
  apply blast
  apply (subst operator_shape_deps_0)
  by blast

subsection ‹Terms›

datatype nterm = 
  VarApp variable "nterm list" 
| AbsApp abstraction "variable list" "nterm list"

definition xvar :: variable ("`x") where "`x = ''x''"
definition xvar0 :: nterm ("§x")  where "§x = VarApp `x []"

definition yvar :: variable ("`y") where "`y = ''y''"
definition yvar0 :: nterm ("§y")  where "§y = VarApp `y []"

definition Avar :: variable ("`A") where "`A = ''A''"
definition Avar0 :: nterm ("§A")  where "§A = VarApp `A []"
definition Avar1 :: "nterm ⇒ nterm" ("§A[_]")  where "§A[t] = VarApp `A [t]"

definition Bvar :: variable ("`B") where "`B = ''B''"
definition Bvar0 :: nterm ("§B")  where "§B = VarApp `B []"
definition Bvar1 :: "nterm ⇒ nterm" ("§B[_]")  where "§B[t] = VarApp `B [t]"

definition Cvar :: variable ("`C") where "`C = ''C''"
definition Cvar0 :: nterm ("§C")  where "§C = VarApp `C []"
definition Cvar1 :: "nterm ⇒ nterm" ("§C[_]")  where "§C[t] = VarApp `C [t]"

definition implies_app :: "nterm ⇒ nterm ⇒ nterm" (infix "`⇒" 225) where 
  "A `⇒ B = AbsApp abstr_implies [] [A, B]"  

definition true_app :: nterm ("`⊤") where "`⊤ = AbsApp abstr_true [] []"

definition false_app :: nterm ("`⊥") where "`⊥ = AbsApp abstr_false [] []"

definition forall_app :: "variable ⇒ nterm ⇒ nterm" ("(3`∀_. _)" [1000, 210] 210) where
  "forall_app x P = AbsApp abstr_forall [x] [P]"

subsection ‹Wellformedness›

fun nt_wf :: "signature ⇒ nterm ⇒ bool" where
  "nt_wf sig (VarApp x ts) = (∀ t = ts!_. nt_wf sig t)"
| "nt_wf sig (AbsApp a xs ts) = 
     (sig_contains sig a (length xs) (length ts) ∧
      distinct xs ∧
      (∀ t = ts!_. nt_wf sig t))"

lemma nt_wf_x0[iff]: "nt_wf sig §x" by (simp add: xvar0_def)
lemma nt_wf_y0[iff]: "nt_wf sig §y" by (simp add: yvar0_def)
lemma nt_wf_A0[iff]: "nt_wf sig §A" by (simp add: Avar0_def)
lemma nt_wf_A1[iff]: "nt_wf sig §A[t] = nt_wf sig t" 
  by (simp add: Avar1_def)
lemma nt_wf_B0[iff]: "nt_wf sig §B" by (simp add: Bvar0_def)
lemma nt_wf_B1[iff]: "nt_wf sig §B[t] = nt_wf sig t" 
  by (simp add: Bvar1_def)
lemma nt_wf_C0[iff]: "nt_wf sig §C" by (simp add: Cvar0_def)
lemma nt_wf_C1[iff]: "nt_wf sig §C[t] = nt_wf sig t" 
  by (simp add: Cvar1_def)

lemma nt_wf_true[simp]: "nt_wf 𝔇 `⊤"
  by (simp add: true_app_def)

lemma nt_wf_implies[simp]: "nt_wf 𝔇 (A `⇒ B) = (nt_wf 𝔇 A ∧ nt_wf 𝔇 B)"
  by (auto simp add: implies_app_def shift_index_def)

lemma nt_wf_forall[simp]: "nt_wf 𝔇 (`∀ `x. t) = nt_wf 𝔇 t"
  by (auto simp add: forall_app_def)

lemma sig_extends_nt_wf: "V ≽ U ⟹ nt_wf U t ⟹ nt_wf V t"
proof (induct t)
  case (VarApp x ts)
  then show ?case by simp
next
  case (AbsApp a xs ts)
  then show ?case 
    by (auto simp add: extends_sig_contains)
qed

subsection ‹Free Variables›

fun nt_free :: "signature ⇒ nterm ⇒ variables" where
  "nt_free sig (VarApp x ts) = 
     (§fold X = {(x, length ts)}, t = ts!_. X ∪ nt_free sig t)" 
| "nt_free sig (AbsApp a xs ts) = 
     (§fold X = {}, t = ts!i. X ∪ (nt_free sig t - (sig!!a.@i(xs))`,0))"

lemma nt_free_x0: "nt_free sig §x = {(`x, 0)}" by (simp add: xvar0_def)
 
lemma nt_free_y0: "nt_free sig §y = {(`y, 0)}" by (simp add: yvar0_def)

lemma nt_free_A0:  "nt_free sig §A = {(`A, 0)}" by (simp add: Avar0_def)
lemma nt_free_A1:  "nt_free sig §A[t] = {(`A, Suc 0)} ∪ nt_free sig t" 
  by (simp add: Avar1_def)

lemma nt_free_B0:  "nt_free sig §B = {(`B, 0)}" by (simp add: Bvar0_def)
lemma nt_free_B1:  "nt_free sig §B[t] = {(`B, Suc 0)} ∪ nt_free sig t" 
  by (simp add: Bvar1_def)

lemma nt_free_C0:  "nt_free sig §C = {(`C, 0)}" by (simp add: Cvar0_def)
lemma nt_free_C1:  "nt_free sig §C[t] = {(`C, Suc 0)} ∪ nt_free sig t" 
  by (simp add: Cvar1_def)

lemma nt_free_true: "nt_free 𝔇 `⊤ = {}"
  by (simp add: true_app_def)

lemma nt_free_implies: "nt_free 𝔇 (s `⇒ t) = nt_free 𝔇 s ∪ nt_free 𝔇 t"
  by (auto simp add: implies_app_def shift_index_def)

lemma nt_free_forall: "nt_free 𝔇 (`∀ x. t) = nt_free 𝔇 t - {(x, 0)}"
  thm forall_app_def binders_as_vars_def
  apply (subst forall_app_def)
  apply auto
  apply (auto simp add: binders_as_vars_def)
  using deduction_sig_forall has_shape_get by auto

lemma sig_extends_nt_free: "V ≽ U ⟹ nt_wf U t ⟹ nt_free V t = nt_free U t"
proof(induct t)
  case (VarApp x ts)
  then show ?case 
    apply simp
    apply (subst list_indexed_fold_cong)
    using VarApp
    by auto
next
  case (AbsApp a xs ts)
  let ?F = "λ i X t. X ∪ (nt_free V t - (V!!a.@i(xs))`,0)"
  let ?G = "λ i X t. X ∪ (nt_free U t - (U!!a.@i(xs))`,0)"
  show ?case 
    apply simp
    thm list_indexed_fold_eq[where ?F = ?F and ?G = ?G]
    apply (subst list_indexed_fold_eq[where ?F = ?F and ?G = ?G])
    using AbsApp
    apply auto
    apply (metis (no_types, lifting) extends_sig_def map_forced_get_def option.case_eq_if 
           sig_contains_def)
    by (metis (no_types, lifting) extends_sig_def map_forced_get_def option.case_eq_if 
        sig_contains_def)
qed
 
lemma nt_free_VarApp: "nt_free sig (VarApp x ts) =
  {(x, length ts)} ∪ ⋃ { nt_free sig t | t. t ∈ set ts }"
proof - 
  have "nt_free sig (VarApp x ts) = (§fold X = {(x, length ts)}, t = ts!_. X ∪ nt_free sig t)"
    by simp
  moreover have "(§fold X = {(x, length ts)}, t = ts!_. X ∪ nt_free sig t) =
         {(x, length ts)} ∪ ⋃ { nt_free sig t | t. t ∈ set ts }"
    by (subst union_unindexed_fold, simp)
  ultimately show ?thesis by simp
qed

lemma nt_free_VarApp_arg_subset:
  assumes "nt_free sig (VarApp x ts) ⊆ X"
  assumes "i < length ts" 
  shows "nt_free sig (ts ! i) ⊆ X"
  by (smt (verit, best) CollectI assms(1) assms(2) le_supE mem_simps(9) nt_free_VarApp nth_mem subset_iff)

lemma nt_free_ConsApp: 
  shows "nt_free sig (AbsApp a xs ts) = 
    ⋃ { nt_free sig (ts!i) - (sig!!a.@i(xs))`,0 | i. i < length ts }"
  by (simp add: union_indexed_fold)

lemma nt_free_ConsApp_arg_subset:
  assumes "nt_free sig (AbsApp a xs ts) ⊆ X"
  assumes "i < length ts"
  shows "nt_free sig (ts!i) ⊆ X ∪ (sig!!a.@i(xs))`,0"
proof - 
  have "nt_free sig (ts!i) - (sig!!a.@i(xs))`,0 ⊆ X"
    by (smt (verit, del_insts) CollectI assms(1) assms(2) mem_simps(9) nt_free_ConsApp subset_eq)
  then show ?thesis by blast
qed
