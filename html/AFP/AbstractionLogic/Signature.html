Theory Signature

theory Signature imports Shape
begin

section ‹Signature›

subsection ‹Abstractions›

datatype abstraction = Abs string

definition abstr_true :: abstraction where "abstr_true = Abs ''true''"
definition abstr_implies :: abstraction where "abstr_implies = Abs ''implies''"
definition abstr_forall :: abstraction where "abstr_forall = Abs ''forall''"
definition abstr_false :: abstraction where "abstr_false = Abs ''false''"

lemma noteq_abstr_true_implies[simp]: "abstr_true ≠ abstr_implies"
  by (simp add: abstr_implies_def abstr_true_def)

lemma noteq_abstr_implies_forall[simp]: "abstr_implies ≠ abstr_forall"
  by (simp add: abstr_implies_def abstr_forall_def)

lemma noteq_abstr_true_forall[simp]: "abstr_true ≠ abstr_forall"
  by (simp add: abstr_true_def abstr_forall_def)

subsection ‹Signatures›

type_synonym signature = "(abstraction, shape) map"

definition empty_sig :: signature where 
  "empty_sig = (λ a. None)"

definition has_shape :: "signature ⇒ abstraction ⇒ shape ⇒ bool" where
  "has_shape S a shape = (S a = Some shape)"
  
definition extends_sig :: "signature ⇒ signature ⇒ bool" (infix "≽" 50) where
  "extends_sig T S = (∀ a. S a = None ∨ T a = S a)"

lemma has_shape_extends: "T ≽ S ⟹ has_shape S a s ⟹ has_shape T a s"
  by (metis extends_sig_def has_shape_def option.discI)

definition sig_contains :: "signature ⇒ abstraction ⇒ nat ⇒ nat ⇒ bool" where
  "sig_contains sig abstr valence arity = 
     (case sig abstr of 
        Some s ⇒ §val s = valence ∧ §ar s = arity 
      | None ⇒ False)"

lemma has_shape_sig_contains: "has_shape sig a s ⟹ sig_contains sig a (§val s) (§ar s)"
  by (simp add: has_shape_def sig_contains_def)

lemma has_shape_get: "has_shape sig a s ⟹ sig !! a = s"
  by (simp add: has_shape_def map_forced_get_def)

lemma extends_sig_contains: "V ≽ U ⟹ sig_contains U a val ar ⟹ sig_contains V a val ar"
  by (smt (verit) extends_sig_def option.case_eq_if sig_contains_def)

subsection ‹Logic Signatures›

definition deduction_sig :: signature ("𝔇") where 
  "𝔇 = empty_sig(
     abstr_true := Some value_shape,
     abstr_implies := Some binop_shape,
     abstr_forall := Some operator_shape)"

lemma deduction_sig_true[iff]: "has_shape deduction_sig abstr_true value_shape"
  by (simp add: has_shape_def deduction_sig_def)

lemma deduction_sig_implies[iff]: "has_shape 𝔇 abstr_implies binop_shape"
  by (simp add: has_shape_def deduction_sig_def)

lemma deduction_sig_forall[iff]: "has_shape 𝔇 abstr_forall operator_shape"
  by (simp add: has_shape_def deduction_sig_def)

lemma deduction_sig_contains_true[iff]: "sig_contains 𝔇 abstr_true 0 0"
  by (simp add: deduction_sig_def sig_contains_def)

lemma deduction_sig_contains_implies[iff]: "sig_contains 𝔇 abstr_implies 0 (Suc (Suc 0))"
  by (simp add: deduction_sig_def sig_contains_def)

lemma deduction_sig_contains_forall[iff]: "sig_contains 𝔇 abstr_forall (Suc 0) (Suc 0)"
  using has_shape_sig_contains deduction_sig_forall by fastforce  





end
